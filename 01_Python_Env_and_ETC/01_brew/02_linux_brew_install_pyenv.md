# install and configure pyenv
```
$ brew install pyenv

$ which pyenv
/home/linuxbrew/.linuxbrew/bin/pyenv

$ pyenv versions
* system (set by /home/ubuntu/.pyenv/version)

$ vi ~/.bashrc
...
# pyenv
export PYENV_ROOT="$HOME/.pyenv"
export PATH="$PYENV_ROOT/bin:$PATH"
eval "$(pyenv init --path)"
eval "$(pyenv init -)"

$ source ~/.bashrc
```
